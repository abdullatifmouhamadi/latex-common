{
\pgfdecoration{{zigzag}{\pgfdecoratedpathlength}{%Setting up the decoration for the full path
                        \pgfdecorationsegmentlength=4mm      % The zigzag thread width
                        \pgfdecorationsegmentamplitude=1mm   % The zigzag thread height
                   }
              }
