\newcommand*\SectionNumberBox[2]
  {%
    \colorbox{#2}
      {%
        \makebox[2.5em][c]
          {%
            \color{white}%
            \strut
            \csname the#1\endcsname
          }%
      }%
    \TitleUnderline{\ \ \ }%
  }

\newcommand*\chapterlabel{}
\titleformat{\chapter}
  {%
    %\thispagestyle{chapterstart}
    \pagecolor{white}
    \bfseries\LARGE\raggedleft
    \color{general@perso@purple}
  }{~}
  {0pt}
  {\formatChapter}
%\titlespacing*{\chapter}{0pt}{50pt}{-60pt}


\titleformat{\section}
  {\Large\bfseries\sffamily\color{general@best@fuchsia}}
  {\SectionNumberBox{section}{general@best@fuchsia}}
  {0pt}
  {\formatSection}
\titleformat{\subsection}
  {\large\bfseries\sffamily\color{general@best@blue}}
  {\SectionNumberBox{subsection}{general@best@blue}}
  {0pt}
  {\formatSubsection}


  %https://tex.stackexchange.com/questions/160320/customizing-chapter-style-with-tikz
  %http://www.texample.net/tikz/examples/fancy-chapter-headings/
  \newcommand{\formatChapter}[1]{%
    \begin{tikzpicture}[remember picture,overlay]
      \node[yshift=-3cm] at (current page.north west)
        {\begin{tikzpicture}[remember picture, overlay]
          \draw[fill=general@perso@purple] (0,0) rectangle
            (\paperwidth,3cm);
          \node[anchor=east,xshift=.9\paperwidth,rectangle,
                rounded corners=2pt,inner sep=11pt,
                fill=general@perso@purple!80]
                {\color{white}\chapterlabel#1};
        \end{tikzpicture}
        };
    \end{tikzpicture}
  }
  \newcommand{\formatSection}[1]{%
    \TitleUnderline{#1}
  }
  \newcommand{\formatSubsection}[1]{%
    \TitleUnderline{#1}
  }

\def\contentsline#1#2#3#4{%
  \ifx\\#4\\%
    \csname l@#1\endcsname{#2}{#3}%
  \else
      \csname l@#1\endcsname{\hyper@linkstart{link}{#4}{#2}\hyper@linkend}{%
        \hyper@linkstart{link}{#4}{#3}\hyper@linkend
      }%
  \fi
}
